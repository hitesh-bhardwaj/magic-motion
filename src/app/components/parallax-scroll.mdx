import { useEffect, useRef } from "react";
import gsap from "gsap";

export default function ParallaxScroll() {
  const ref = useRef();
  useEffect(() => {
    gsap.to(ref.current, { y: 100, scrollTrigger: ref.current });
  }, []);
  return <div ref={ref}>Parallax Me!</div>;
}
