import { useEffect, useRef } from "react";
import gsap from "gsap";

export default function FadeInText() {
  const ref = useRef();
  useEffect(() => {
    gsap.fromTo(ref.current, { opacity: 0 }, { opacity: 1, duration: 1.2 });
  }, []);
  return <div ref={ref}>Hello, GSAP Animation!</div>;
}
